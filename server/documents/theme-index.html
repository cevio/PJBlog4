处理日志列表,在列表区域加入以下代码

<%
;(function(lists, pages){		
	for ( var i = 0 ; i < lists.length ; i++ ){
		/*
			日志列表数据结构
			{
				id: this("id").value, 日志ID
				title: this("log_title").value, 日志标题
				postDate: this("log_posttime").value, 日志发表时间
				editDate: this("log_updatetime").value, 日志修改时间
				category: getCategoryName(this("log_category").value), 日志分类信息
						日志分类信息包含以下内容
						{
							id: id, 分类ID
							name: name, 分类名称
							info: info, 分类说明
							icon: icon, 分类ICON
							url: url 分类的连接地址
						}
				tags: getTags(this("log_tags").value), 日志标签信息
						日志标签信息包含以下内容
						{
							id: id, 标签ID
							name: name, 标签名
							url: url, 标签连接地址
							count: count 标签个数
						}
				content: this("log_shortcontent").value, 日志内容
				url: "article.asp?id=" + this("id").value, 日志连接地址
				views: this("log_views").value, 日志查看次数
				uid: getUserPhoto(this("log_uid").value), 发表日志用户信息
						日志用户信息包含以下内容
						{
							photo: photo, 用户头像
							id: id, 用户ID -1为超级管理员
							name: name, 用户名
							poster: poster, 是否具有poster权限
							oauth: oauth, 登入方式 qq or other
							login: login, 是否已经登入 true or false
							logindate: logindate, 登入时间
							loginip: loginip 登入IP
						}
				istop: this("log_istop").value, 日志是否为置顶状态
				cover: this("log_cover").value 日志封面
			}
			以下展示不完整，你可以自己来安排展示内容。通过对以上变量的操作。
		*/
%>
	<div class="article-content">
		<h1><a href="<%=lists[i].url%>"><%=lists[i].title%></a></h1>
		<div class="pj-article-infos">
        	<span></span>
			<span class="date"><%=date.format(lists[i].postDate, "M d y - h:i")%></span>
			<span class="tags"><%
				for ( j = 0 ; j < lists[i].tags.length ; j++ ){
%>
				<a href="<%=lists[i].tags[j].url%>"><%=lists[i].tags[j].name%></a>
<%
				}
%>			</span>
		</div>
		<div class="pj-content"><%=lists[i].content%></div>
    </div>
<%
	}
		
		// 处理分页
	if ( pages.length > 0 ){
%>
		<div class="pj-article-pagebar fn-clear">
<%
		for ( i = 0 ; i < pages.length ; i++ ){
				/*
					数据结构如下
					[
						{key: 1, url: "..."},
						{key: 2, url: "...."}
						......
					]
					key是页数
					url是连接地址
				*/
			if ( pages[i].url === undefined ){
%>
			<span class="fn-left"><%=pages[i].key%></span>
<%
			}else{
%>
				<a href="<%=pages[i].url%>" class="fn-left"><%=pages[i].key%></a>
<%			
			}
		}
%>
		</div>		
<%
}
})(pageCustomParams.articles.lists, pageCustomParams.articles.pages);
%>
